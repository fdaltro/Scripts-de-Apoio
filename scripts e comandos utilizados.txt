# Envio de mensagens para Fila

QUEUE_URL="https://sqs.us-east-1.amazonaws.com/504491092699/togglemaster-queue"

while true; do
  aws sqs send-message --queue-url "$QUEUE_URL" \
    --message-body "{
      \"user_id\": \"aws-shell-user\",
      \"flag_name\": \"enable-new-dashboard\",
      \"result\": true,
      \"timestamp\": \"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"
    }" > /dev/null && echo "📥 Mensagem enviada às $(date +%H:%M:%S)"
  sleep 1
done

# Script de erro 400 para disparar o pagerduty
URL="http://localhost:8001"

echo "Iniciando tráfego... Vamos ver o que o servidor responde:"
while true; do
  curl -s -o /dev/null -w "Health: %{http_code} | " -X GET "$URL/health"
  curl -s -o /dev/null -w "Validate: %{http_code}\n" -X POST "$URL/validate" -H "Authorization: Bearer chave-fake" -H "Content-Type: application/json"
  sleep 0.5
done


#Script fr health para gerar traces dos serviços

while true; do
  echo "--- Enviando telemetria $(date +%H:%M:%S) ---"
  for port in 8001 8002 8003 8004 8005; do
    # Faz o curl e extrai apenas o código HTTP
    status=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:$port/health || echo "OFF")
    echo "Porta $port: $status"
  done
  sleep 0.1 # Espera 5 segundos antes da próxima rodada
done
